<script>
    let familyData = JSON.parse(localStorage.getItem('familyFinalDB_2024')) || [];
    let draggedId = null;

    // --- Switch Views ---
    function switchView(view) {
        const album = document.getElementById('albumView');
        const tree = document.getElementById('treeView');
        document.querySelectorAll('.nav-btn').forEach(b => b.classList.remove('active'));

        if (view === 'album') {
            album.style.display = 'grid';
            tree.style.display = 'none';
            document.getElementById('btnAlbum').classList.add('active');
            renderAlbum();
        } else {
            album.style.display = 'none';
            tree.style.display = 'block';
            document.getElementById('btnTree').classList.add('active');
            renderTree();
        }
    }

    // --- Modal Logic ---
    function openModal(id = null) {
        document.getElementById('memberModal').style.display = 'flex';
        if (id) {
            const m = familyData.find(x => x.id === id);
            document.getElementById('editId').value = m.id;
            document.getElementById('mName').value = m.name;
            document.getElementById('mRel').value = m.relation;
            document.getElementById('modalTitle').innerText = "‡πÅ‡∏Å‡πâ‡πÑ‡∏Ç‡∏Ç‡πâ‡∏≠‡∏°‡∏π‡∏•";
        } else {
            document.getElementById('editId').value = '';
            document.getElementById('mName').value = '';
            document.getElementById('mRel').value = '';
            document.getElementById('modalTitle').innerText = "‡πÄ‡∏û‡∏¥‡πà‡∏°‡∏™‡∏°‡∏≤‡∏ä‡∏¥‡∏Å‡πÉ‡∏´‡∏°‡πà";
        }
    }

    function closeModal() { document.getElementById('memberModal').style.display = 'none'; }

    // --- Save & Storage ---
    async function saveMember() {
        const id = document.getElementById('editId').value || Date.now().toString();
        const name = document.getElementById('mName').value;
        const relation = document.getElementById('mRel').value;
        const file = document.getElementById('mImg').files[0];

        if (!name) return alert("‡πÇ‡∏õ‡∏£‡∏î‡∏£‡∏∞‡∏ö‡∏∏‡∏ä‡∏∑‡πà‡∏≠‡∏Ñ‡∏£‡∏±‡∏ö");

        let imgData = "https://www.w3schools.com/howto/img_avatar.png";
        const existing = familyData.find(m => m.id === id);
        if (existing) imgData = existing.img;

        if (file) {
            imgData = await new Promise(res => {
                const rd = new FileReader();
                rd.onload = () => res(rd.result);
                rd.readAsDataURL(file);
            });
        }

        const newMember = { id, name, relation, img: imgData, parentId: existing ? existing.parentId : "" };
        const idx = familyData.findIndex(m => m.id === id);
        if (idx > -1) familyData[idx] = newMember;
        else familyData.push(newMember);

        syncData();
        closeModal();
    }

    function deleteMember(id) {
        if(confirm("‡∏•‡∏ö‡∏Ç‡πâ‡∏≠‡∏°‡∏π‡∏•‡∏ô‡∏µ‡πâ ‡∏Ç‡πâ‡∏≠‡∏°‡∏π‡∏•‡∏•‡∏π‡∏Å‡∏´‡∏•‡∏≤‡∏ô‡πÉ‡∏ô‡πÅ‡∏ú‡∏ô‡∏ú‡∏±‡∏á‡∏à‡∏∞‡∏´‡∏≤‡∏¢‡πÑ‡∏õ‡∏î‡πâ‡∏ß‡∏¢ ‡∏¢‡∏∑‡∏ô‡∏¢‡∏±‡∏ô‡πÑ‡∏´‡∏°‡∏Ñ‡∏£‡∏±‡∏ö?")) {
            familyData = familyData.filter(m => m.id !== id);

    }

    function syncData() {
        localStorage.setItem('familyFinalDB_2024', JSON.stringify(familyData));
        renderAlbum();
        renderTree();
    }

    // --- Render Album ---
    function renderAlbum() {
        const grid = document.getElementById('albumView');
        grid.innerHTML = '';

        });
    }

    // --- Render Tree & Drag/Drop ---
    function renderTree() {
        document.getElementById('treeContent').innerHTML = buildTreeHTML("");
    }

    function buildTreeHTML(pId = "") {
        const children = familyData.filter(m => m.parentId === pId);
        if (children.length === 0) return "";
        let html = "<ul>";
        children.forEach(m => {
            html += `
                <li>
                    <div class="node" draggable="true" 
                         ondragstart="draggedId='${m.id}'"
                         ondragover="e=>{e.preventDefault(); e.currentTarget.classList.add('drag-over')}"
                         ondragleave="e=>e.currentTarget.classList.remove('drag-over')"
                         ondrop="dropOnMember(event, '${m.id}')">
                        <img src="${m.img}">
                        <span>${m.name}</span>
                        <small style="font-size:0.7rem; color:var(--gold)">${m.relation}</small>
                    </div>
                    ${buildTreeHTML(m.id)}
                </li>`;
        });
        html += "</ul>";
        return html;
    }

    function dropOnMember(e, targetId) {
        e.preventDefault();
        e.currentTarget.classList.remove('drag-over');
        if (draggedId === targetId) return;
        const member = familyData.find(m => m.id === draggedId);
        member.parentId = targetId;
        syncData();
    }

    function dropToRoot(e) {
        const member = familyData.find(m => m.id === draggedId);
        member.parentId = "";
        syncData();
    }

    // --- Export / Import ---
