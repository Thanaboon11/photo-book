<script>
    let members = JSON.parse(localStorage.getItem('familyTreeData')) || [];

    function openModal(id = null) {
        const modal = document.getElementById('memberModal');
        const parentSelect = document.getElementById('mParent');
        document.getElementById('editId').value = id ? id : '';
        
        // อัปเดตรายชื่อพ่อแม่ใน Dropdown
        parentSelect.innerHTML = '<option value="">-- เป็นระดับสูงสุด (ไม่มีพ่อแม่ในผัง) --</option>';
        members.forEach(m => {
            if (m.id !== id) {
                parentSelect.innerHTML += `<option value="${m.id}">${m.name}</option>`;
            }
        });

        if (id) {
            const m = members.find(item => item.id === id);
            document.getElementById('mName').value = m.name;
            document.getElementById('mRel').value = m.relation;
            document.getElementById('mParent').value = m.parentId || '';
        } else {
            document.getElementById('mName').value = '';
            document.getElementById('mRel').value = '';
            document.getElementById('mParent').value = '';
        }
        modal.style.display = 'flex';
    }

    function closeModal() {
        document.getElementById('memberModal').style.display = 'none';
    }

    async function saveMember() {
        const id = document.getElementById('editId').value || Date.now().toString();
        const name = document.getElementById('mName').value;
        const relation = document.getElementById('mRel').value;
        const parentId = document.getElementById('mParent').value;
        const fileInput = document.getElementById('mImg');

        if (!name) return alert("โปรดใส่ชื่อด้วยครับ");

        let imgBase64 = "https://via.placeholder.com/100";
        const existing = members.find(m => m.id === id);
        if (existing) imgBase64 = existing.img;

        if (fileInput.files && fileInput.files[0]) {
            imgBase64 = await convertBase64(fileInput.files[0]);
        }

        const memberData = { id, name, relation, parentId, img: imgBase64 };

        const index = members.findIndex(m => m.id === id);
        if (index > -1) members[index] = memberData;
        else members.push(memberData);

        localStorage.setItem('familyTreeData', JSON.stringify(members));
        renderTree();
        closeModal();
    }

    function convertBase64(file) {
        return new Promise((resolve, reject) => {
            const reader = new FileReader();
            reader.readAsDataURL(file);
            reader.onload = () => resolve(reader.result);
            reader.onerror = error => reject(error);
        });
    }

    function deleteMember(id) {
        if (confirm("ลบสมาชิกคนนี้ แผนผังลูกหลานอาจจะหายไปด้วย ยืนยันไหมครับ?")) {
            members = members.filter(m => m.id !== id);
            localStorage.setItem('familyTreeData', JSON.stringify(members));
            renderTree();
        }
    }

    // ฟังก์ชันสร้างโครงสร้าง Tree แบบ Recursive
    function buildTreeHTML(parentId = "") {
        const children = members.filter(m => m.parentId === parentId);
        if (children.length === 0) return "";

        let html = "<ul>";
        children.forEach(m => {
            html += `
                <li>
                    <div class="member-node">
                        <div class="photo-box"><img src="${m.img}"></div>
                        <div class="name">${m.name}</div>
                        <div class="rel">${m.relation}</div>
                        <div class="controls">
                            <button class="btn-mini" onclick="openModal('${m.id}')">แก้ไข</button>
                            <button class="btn-mini" style="color:red" onclick="deleteMember('${m.id}')">ลบ</button>
                        </div>
                    </div>
                    ${buildTreeHTML(m.id)}
                </li>
            `;
        });
        html += "</ul>";
        return html;
    }

    function renderTree() {
        const container = document.getElementById('familyTree');
        container.innerHTML = buildTreeHTML("");
    }

    renderTree();
</script>
